#BEGIN 0000H
  
    LXI SP,0FFFFH

INIT:   MVI A,88H  ;8255 initialiaztion with upper port C input
        OUT 83H
        MVI A,38H  ;LCD intilaization
        CALL WR_CMD
        MVI A,38H   ;LCD intilaization
        CALL WR_CMD
        MVI A,38H   ;LCD intilaization
        CALL WR_CMD

SCREEN1:  MVI A,01H     ;LCD intilaization
          CALL WR_CMD
          MVI A,0CH    ;LCD intilaization
          CALL WR_CMD
          MVI A,80H   ;Cursor moving to top left adress of lcd
          CALL WR_CMD 
           MVI A ,44H    ; "D"
           CALL WR_CHR
           MVI A,45H      ;"E"
           CALL WR_CHR
           MVI A,4CH      ; "L"
           CALL WR_CHR
           MVI A,41H 
           CALL WR_CHR
           MVI A,59H      ;"Y"
           CALL WR_CHR  
           MVI A,3AH      ;"
           CALL WR_CHR
         
           MVI A,C0H      ;cursor to 2nd line left
           CALL WR_CMD
           MVI A ,30H
           CALL WR_CHR
           MVI A ,30H
           CALL WR_CHR
           MVI A,30H 
           CALL WR_CHR
           MVI A,73H 
           CALL WR_CHR
           
          JMP BUTTON

WR_CMD:   OUT 81H
        MVI A,01H
        OUT 82H
       CALL DELAY1
        MVI A,00H
        OUT 82H
        CALL DELAY1
        RET
        
WR_CHR: OUT 81H
        MVI A,03H
        OUT 82H
       CALL DELAY1       
        MVI A,02H
        OUT 82H
        CALL DELAY1
        RET

DELAY1:PUSH B
        MVI C,8CH
LOOP_1: DCR C
        JNZ LOOP_1
        POP B
        RET

DELAY2:PUSH B
        MVI C,4CH
LOOP_2: DCR C
        JNZ LOOP_2
        POP B
        RET


BUTTON:MVI B,00H 
BACK:IN 82H
    CALL DELAY2
    CPI B2H
    JZ NEXTSTEP
    CPI E2H
   JZ INCREASE
    CPI D2H
    JNZ BACK
    MOV A,B
    CPI 00H
    JZ ERROR1
    DCR B
    MVI A,C0H
    CALL WR_CMD ;"cursor next line"
    MOV A,B
    CALL DEKHA
    WAIT:IN 82H
    CPI D2H
    JZ WAIT
    JMP BACK

INCREASE: INR B
    MVI A,C0H
    CALL WR_CMD ;"cursor next line"
    MOV A,B
    CALL DEKHA
    WAIT_1:IN 82H
    CPI E2H
    JZ WAIT_1
JMP BACK

ERROR1:CALL ERROR
    CALL DELAY_0_5S
    JMP SCREEN1

DEKHA:PUSH B
      PUSH D
      MVI C, 0AH
      MVI D,00H
      MVI E,00H
      MVI L,64H
      MVI H,00H
STEP1: CMP L               
       JC STEP2               
       SUB L                        
       INR H  ;"H holds HUNDREDS place"                  
       JMP STEP1  
STEP2: CMP C                
       JC STEP3                
       SUB C                        
       INR D   ;"D holds tens place"                  
       JMP STEP2                
STEP3:MOV E,A  ;"E holds ones place"
      MOV A,H
      ADI 30H
      CALL WR_CHR
      MOV A,D
      ADI 30H  ;"ASCII Conversion"
      CALL WR_CHR
      MOV A,E
      ADI 30H
      CALL WR_CHR
      POP D
      POP B
RET

NEXTSTEP: MOV A,B
                ADI 00H
  CZ ERROR
  JZ INIT
SCREEN2:       MVI A,01H;"no of shots screen"
           CALL WR_CMD   
           MVI A,80H
          CALL WR_CMD
          MVI A ,53H    ; "S"
           CALL WR_CHR
           MVI A,48H      ;"h"
           CALL WR_CHR
           MVI A,4FH      ; "o"
           CALL WR_CHR
           MVI A,54H      ;"T"
           CALL WR_CHR
           MVI A,53H      ;"S"
           CALL WR_CHR  
           MVI A,28H      ;"("
           CALL WR_CHR   
           MVI A,30H      ;"0"
           CALL WR_CHR  
          MVI A,20H      ;" "
           CALL WR_CHR  
           MVI A,46H      ;"F "
           CALL WR_CHR  
            MVI A,4FH      ;"o"
           CALL WR_CHR  
           MVI A,52H      ;"R"
           CALL WR_CHR  
          MVI A,20H      ;" "
           CALL WR_CHR  
           MVI A,3EH      ;"~"
           CALL WR_CHR
          MVI A,3CH      ;"~"
           CALL WR_CHR
           MVI A,29H      ;")"
           CALL WR_CHR   
           MVI A,3AH      ;
           CALL WR_CHR
           MVI A,C0H      ;cursor to 2nd line left
          CALL WR_CMD
           MVI A,30H      ;"0"
           CALL WR_CHR
          MVI A ,30H
           CALL WR_CHR
          MVI A ,30H      ;"0"
           CALL WR_CHR
          CALL DELAY_0_5S
           MVI C,00H 
BACK1:IN 82H
      CALL DELAY2
      CPI B2H
      JZ NEXT
      CPI E2H
       JZ INCREASE1
    CPI D2H
    JNZ BACK1
 MOV A,C
    CPI 00H
    CZ ERROR2
    DCR C
    MVI A,C0H
    CALL WR_CMD ;"cursor next line"
    MOV A,C
    CALL DEKHA
    WAIT1:IN 82H
    CPI D2H
    JZ WAIT1
    JMP BACK1

INCREASE1: INR C
    MVI A,C0H
    CALL WR_CMD ;"cursor next line"
    MOV A,C
    CALL DEKHA
    WAIT_1_1:IN 82H
    CPI E2H
    JZ WAIT_1_1
JMP BACK1

ERROR2:CALL ERROR
    CALL DELAY_0_5S
    JMP SCREEN2

NEXT:CALL RUNNING; "Shows Running screen"
    MOV E,B ;"Backup"
    MOV A,C
    ADI 00H
    JNZ LOOP

INFINITY: MOV B,E         ;TRAPPED HERE
          CALL SECONDS
          MOV B,E
          CALL SECONDS
          CALL SHOW
          JMP INFINITY

LOOP:MOV B,E
CALL SECONDS
MOV B,E
CALL SECONDS
CALL SHOW
DCR C
JNZ LOOP
JMP INIT


SHOW:    MVI A,01H   ;Focus first
         OUT 80H      ;Port A
        CALL DELAY_CAM
         MVI A,00H      
         OUT 80H
         RET

SECONDS:  CALL DELAY_0_5S
          DCR B
          JNZ SECONDS
          RET

DELAY_CAM: PUSH D
            LXI D,4E20H
      DEL1:DCX D
          MOV A,D
          ORA E
          JNZ DEL1
          POP D
          RET

DELAY_0_5S: PUSH D
            LXI D,A2C2H
      DEL:DCX D
          MOV A,D
          ORA E
          JNZ DEL
          POP D
          RET

RUNNING:MVI A,01H
           CALL WR_CMD   
           MVI A,80H
          CALL WR_CMD
          MVI A,52H 
          CALL WR_CHR
          MVI A,55H 
           CALL WR_CHR
          MVI A,4EH 
          CALL WR_CHR
          MVI A,4EH 
          CALL WR_CHR
          MVI A,49H 
          CALL WR_CHR
          MVI A,4EH 
          CALL WR_CHR
          MVI A,47H 
          CALL WR_CHR
          RET

ERROR:     MVI A,01H
           CALL WR_CMD   
           MVI A,80H
           CALL WR_CMD
           MVI A,45H 
           CALL WR_CHR
           MVI A,52H 
           CALL WR_CHR
           MVI A,52H 
           CALL WR_CHR
           MVI A,4FH 
           CALL WR_CHR
           MVI A,52H 
           CALL WR_CHR
           CALL DELAY_0_5S
           RET

LIMITERROR:MVI A,01H
           CALL WR_CMD   
           MVI A,80H
          CALL WR_CMD
          MVI A,45H 
          CALL WR_CHR
          MVI A,52H 
          CALL WR_CHR
          MVI A,52H 
           CALL WR_CHR
           MVI A,4FH 
           CALL WR_CHR
           MVI A,52H 
           CALL WR_CHR
           CALL DELAY_0_5S
  RET

           
STOP:HLT  
#END
